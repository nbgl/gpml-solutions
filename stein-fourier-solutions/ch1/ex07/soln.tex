  \begin{align*}
    &A \cos(ct - \varphi) \\
    &= A(\cos ct\cos(-\varphi)-\sin ct\sin(-\varphi))
      &\text{(by 4(i))} \\
    &= A(\cos ct\cos\varphi + \sin ct\sin\varphi)
      &\text{($\cos$ is even and $\sin$ is odd)} \\
    &= \sqrt{a^2+b^2}(\frac{a}{\sqrt{a^2+b^2}}\cos ct
        + \frac{b}{\sqrt{a^2+b^2}}\sin ct)
      &\text{(substitution)}\\
    &= a\cos ct + b\sin ct \text{.}
  \end{align*} \qed
